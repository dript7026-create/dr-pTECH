#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <io.h>
#include <sys/stat.h>
#include <direct.h>

#define MAX_CMD 4096
#define MAX_PATHS 1024

int ends_with_ci(const char *s, const char *ext) {
    if (!s || !ext) return 0;
    size_t ls = strlen(s), le = strlen(ext);
    if (le > ls) return 0;
    const char *p = s + ls - le;
    for (size_t i = 0; i < le; ++i) {
        char a = p[i], b = ext[i];
        if (a >= 'A' && a <= 'Z') a += 'a' - 'A';
        if (b >= 'A' && b <= 'Z') b += 'a' - 'A';
        if (a != b) return 0;
    }
    return 1;
}

int is_image_file(const char *name) {
    const char *exts[] = {".png",".jpg",".jpeg",".bmp",".gif",".tiff", NULL};
    for (int i=0; exts[i]; ++i) if (ends_with_ci(name, exts[i])) return 1;
    return 0;
}

int is_video_file(const char *name) {
    const char *exts[] = {".mp4",".avi",".mov",".mkv",".wmv",".flv", NULL};
    for (int i=0; exts[i]; ++i) if (ends_with_ci(name, exts[i])) return 1;
    return 0;
}

void ensure_dir(const char *path) {
    struct stat st;
    if (stat(path, &st) != 0) {
        _mkdir(path);
    }
}

int run_ffmpeg_extract(const char *video, const char *outdir) {
    char cmd[MAX_CMD];
    ensure_dir(outdir);
    // extract at 10 fps - adjust as needed
    snprintf(cmd, sizeof(cmd),
        "ffmpeg -y -i \"%s\" -vf fps=10 \"%s\\frame_%%04d.png\" >nul 2>nul",
        video, outdir);
    return system(cmd);
}

int write_blender_script(const char *script_path) {
    FILE *f = fopen(script_path, "w");
    if (!f) return 0;
    fprintf(f,
"import bpy, sys, os\n"
"argv = sys.argv\n"
"if '--' in argv:\n"
"    idx = argv.index('--')\n" 
"    media = argv[idx+1]\n"
"    outblend = argv[idx+2]\n"
"else:\n"
"    print('Expected args: -- <media_path> <outblend>')\n" 
"    sys.exit(1)\n"
"\n"
"# start with empty scene\n"
"bpy.ops.wm.read_factory_settings(use_empty=True)\n"
"is_seq = os.path.isdir(media)\n"
"if is_seq:\n"
"    files = sorted([f for f in os.listdir(media) if f.lower().endswith(('.png','.jpg','.jpeg'))])\n" 
"    if not files:\n"
"        print('No frames found in', media); sys.exit(1)\n" 
"    image_path = os.path.join(media, files[0])\n"
"else:\n"
"    image_path = media\n"
"\n"
"# create a plane\n"
"bpy.ops.mesh.primitive_plane_add(size=2, location=(0,0,0))\n"
"obj = bpy.context.active_object\n"
"# subdivide for displacement\n"
"sub = obj.modifiers.new('subd','SUBSURF')\n"
"sub.levels = 2\n"
"# make texture from image\n"
"tex = bpy.data.textures.new('imgTex', type='IMAGE')\n"
"img = bpy.data.images.load(image_path)\n"
"tex.image = img\n"
"# add displacement modifier\n"
"dis = obj.modifiers.new('displace','DISPLACE')\n"
"dis.texture = tex\n"
"dis.strength = 0.5\n"
"# create material with image texture\n"
"mat = bpy.data.materials.new('Mat')\n"
"mat.use_nodes = True\n"
"nodes = mat.node_tree.nodes\n"
"links = mat.node_tree.links\n"
"nodes.clear()\n"
"out = nodes.new('ShaderNodeOutputMaterial')\n"
"bsdf = nodes.new('ShaderNodeBsdfPrincipled')\n"
"img_node = nodes.new('ShaderNodeTexImage')\n"
"img_node.image = img\n"
"links.new(img_node.outputs['Color'], bsdf.inputs['Base Color'])\n"
"links.new(bsdf.outputs['BSDF'], out.inputs['Surface'])\n"
"obj.data.materials.append(mat)\n"
"bpy.ops.object.shade_smooth()\n"
"# save result\n"
"bpy.ops.wm.save_as_mainfile(filepath=outblend)\n"
"print('Saved', outblend)\n");
    fclose(f);
    return 1;
}

int main(int argc, char **argv) {
    const char *indir = ".";
    const char *outdir = ".\\blender_outputs";
    if (argc >= 2) indir = argv[1];
    if (argc >= 3) outdir = argv[2];
    ensure_dir(outdir);

    char search_path[MAX_CMD];
    snprintf(search_path, sizeof(search_path), "%s\\*.*", indir);

    struct _finddata_t fd;
    intptr_t handle = _findfirst(search_path, &fd);
    if (handle == -1) {
        fprintf(stderr, "No files in %s\n", indir);
        return 1;
    }

    char script_path[MAX_CMD];
    snprintf(script_path, sizeof(script_path), "%s\\blender_automation.py", outdir);
    if (!write_blender_script(script_path)) {
        fprintf(stderr, "Failed to write blender script\n");
        return 1;
    }

    do {
        if (fd.attrib & _A_SUBDIR) continue;
        char fullpath[MAX_CMD];
        snprintf(fullpath, sizeof(fullpath), "%s\\%s", indir, fd.name);

        if (is_image_file(fd.name)) {
            char outblend[MAX_CMD];
            snprintf(outblend, sizeof(outblend), "%s\\%s.blend", outdir, fd.name);
            char cmd[MAX_CMD];
            snprintf(cmd, sizeof(cmd),
                "blender --background --python \"%s\" -- \"%s\" \"%s\" >nul 2>nul",
                script_path, fullpath, outblend);
            printf("Processing image: %s\n", fullpath);
            system(cmd);
        } else if (is_video_file(fd.name)) {
            // create frames dir
            char base_noext[512];
            strncpy(base_noext, fd.name, sizeof(base_noext));
            char *dot = strrchr(base_noext, '.'); if (dot) *dot = 0;
            char frames_dir[MAX_CMD];
            snprintf(frames_dir, sizeof(frames_dir), "%s\\%s_frames", outdir, base_noext);
            ensure_dir(frames_dir);
            printf("Extracting frames from: %s -> %s\n", fullpath, frames_dir);
            run_ffmpeg_extract(fullpath, frames_dir);
            // call blender with frames dir
            char outblend[MAX_CMD];
            snprintf(outblend, sizeof(outblend), "%s\\%s.blend", outdir, base_noext);
            char cmd[MAX_CMD];
            snprintf(cmd, sizeof(cmd),
                "blender --background --python \"%s\" -- \"%s\" \"%s\" >nul 2>nul",
                script_path, frames_dir, outblend);
            printf("Processing sequence: %s\n", frames_dir);
            system(cmd);
        } else {
            // skip
        }

    } while (_findnext(handle, &fd) == 0);

    _findclose(handle);
    printf("Done. Outputs in %s\n", outdir);
    return 0;
}
